git remote -v							【查看关联的远程代码库】

git remote rm origin					【删除关联的远程代码库】

git remote add 远程库名 远程库地址		【添加关联的远程代码库】

git push -u 远程库 本地库				【推送并关联远程代码库】

git push 远程库 本地库				   【关联后的推送】



git branch -a                          【查看所有分支】
 
git push origin --delete 远程分支名    【删除远程分支】

git branch -d 本地分支名               【删除本地分支】
